1. التقييم الأمني الأولي

الثغرات المكتشفة

القضية الحرجة: ثغرة في Next.js (CVE-2025-6010) تسمح بكشف البيانات من خلال أخطاء التوجيه.
المشاكل عالية الخطورة: ثغرات متعددة بما في ذلك ReDoS في picomatch والمتجهات المحتملة لحقن SQL.

// src/lib/utils/security.ts import DOMPurify from 'dompurify'; import { escape } from 'html-escaper'; /** * تصفية إدخال النص لمنع هجمات XSS */ export function sanitizeText(input: string | null | undefined): string { if (!input) return ''; // إزالة أي علامات HTML وتصفية الإدخال return DOMPurify.sanitize(input, { ALLOWED_TAGS: [], ALLOWED_ATTR: [] }); } /** * تهريب كيانات HTML لمنع XSS */ export function escapeHtml(input: string): string { return escape(input); }

    3. تحسينات أمن البنية التحتية

    أ. تنفيذ رؤوس الأمان

    السبب: رؤوس الأمان المناسبة تمنع متجهات هجوم مختلفة بما في ذلك النقرات وXSS وفحص نوع المحتوى.
    // src/middleware.ts - رؤوس الأمان export async function middleware(request: NextRequest) { const response = NextResponse.next(); // سياسة أمن المحتوى response.headers.set('Content-Security-Policy', "default-src 'self'; script-src 'self' 'unsafe-eval' 'unsafe-inline'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:;"); // X-Frame-Options response.headers.set('X-Frame-Options', 'SAMEORIGIN'); // X-Content-Type-Options response.headers.set('X-Content-Type-Options', 'nosniff'); // Referrer-Policy response.headers.set('Referrer-Policy', 'strict-origin-when-cross-origin'); return response; }
